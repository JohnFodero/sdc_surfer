
#define esc_pin
#define servo_pin

Servo ESC;
Servo servo;
void setup()
{
	Serial.begin(115200);
	
	ESC.attach(esc_pin);
	servo.attach(servo_pin);
	

	// initialize servo and ESC
	ESC.write(1500);
	servo.write(1500);
	delay(3000);
	
	// get operating mode (0 for capture, 1 for drive)
	while(!Serial.available()) {}
	
}

void loop()
{

}
